rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // âœ… Allow public read access to products (anyone can view)
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // âœ… Allow public read access to categories (anyone can view)
    match /categories/{categoryId} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    
    // âœ… Allow public read access to reviews (anyone can view)
    match /Feedback/{reviewId} {
      allow read: if true;
      allow create: if request.auth != null;
      allow update, delete: if request.auth != null;
    }
    
    // ðŸ”’ Keep orders protected (only authenticated users)
    match /orders/{orderId} {
      allow read, write: if request.auth != null;
    }
    
    // ðŸ”’ Keep users protected (users can only access their own data)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // ðŸ”’ Keep notifications protected (only authenticated users)
    match /notifications/{notificationId} {
      allow read, write: if request.auth != null;
    }
    
    // Default: deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
